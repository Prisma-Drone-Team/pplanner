82 class ClearanceObjective : public ob::StateCostIntegralObjective
83 {
84 public:
85  ClearanceObjective(const ob::SpaceInformationPtr& si) :
86  ob::StateCostIntegralObjective(si, true)
87  {
88  }
89 
90  // Our requirement is to maximize path clearance from obstacles,
91  // but we want to represent the objective as a path cost
92  // minimization. Therefore, we set each state's cost to be the
93  // reciprocal of its clearance, so that as state clearance
94  // increases, the state cost decreases.
95  ob::Cost stateCost(const ob::State* s) const override
96  {
97  return ob::Cost(1 / (si_->getStateValidityChecker()->clearance(s) +
98  std::numeric_limits<double>::min()));
99  }
100 };
